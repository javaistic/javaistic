    }

    // Factorial (for integers)
    public long factorial(int n) {
        if (n < 0) {
            throw new IllegalArgumentException("Factorial is not defined for negative numbers");
        }
        if (n == 0 || n == 1) {
            return 1;
        }

        long result = 1;
        for (int i = 2; i <= n; i++) {
            result *= i;
        }
        return result;
    }

    // GCD using Euclidean algorithm
    public int gcd(int a, int b) {
        a = Math.abs(a);
        b = Math.abs(b);

        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }

    // LCM
    public int lcm(int a, int b) {
        return Math.abs(a * b) / gcd(a, b);
    }

                    activeTab === tab.path
                      ? "bg-zinc-900 text-white border-b-2 border-b-blue-500"
                      : "bg-zinc-800 text-zinc-300 hover:bg-zinc-700"
                  }`}
                  onClick={() => switchTab(tab.path)}
                >
                  {getFileIcon(tab.fileName)}
                  <span className="text-sm truncate">{tab.fileName}</span>
                  <button
                    onClick={(e) => closeTab(tab.path, e)}
                    className="text-zinc-500 hover:text-white text-xs"
                  >
                    <XIcon size={12} />
                  </button>
                </div>
              ))}
            </div>

            {/* Toolbar */}
            <div className="flex items-center space-x-2 px-4 py-2 border-l border-zinc-700 bg-zinc-800 shrink-0">
              <button
                onClick={handleRun}
                disabled={!activeTab}
                className="flex items-center space-x-1 bg-green-600 hover:bg-green-700 disabled:bg-zinc-600 disabled:cursor-not-allowed px-3 py-1 rounded-md text-sm transition-colors"
              >
                <Play size={14} />
                <span>Run</span>
              </button>
              <button
                onClick={handleSave}
                disabled={!activeTab}
                className="flex items-center space-x-1 bg-blue-600 hover:bg-blue-700 disabled:bg-zinc-600 disabled:cursor-not-allowed px-3 py-1 rounded-md text-sm transition-colors"
              >
                <Save size={14} />
                <span>Save</span>
              </button>
            </div>
          </div>
          {/* Code Editor */}
          {activeTab ? (
            <div className="flex-1 flex">
              {/* Line Numbers */}
              {/* <div className="bg-zinc-800 px-3 py-4 text-zinc-500 text-sm font-mono border-r border-zinc-700 min-w-12 select-none">
                {Array.from({ length: lineCount }, (_, i) => (
                  <div key={i + 1} className="leading-6 text-right">
                    {i + 1}
                  </div>
                ))}
              </div> */}

              {/* Code Area */}
              <div className="flex-1 relative">
                {/* Syntax highlighted background */}
                {/* <div
                  className="absolute inset-0 p-4 overflow-hidden font-mono text-sm text-left leading-6 pointer-events-none whitespace-pre-wrap"
                  dangerouslySetInnerHTML={{ __html: syntaxHighlight(code) }}
                /> */}

                {/* Actual textarea */}
                <textarea
                  ref={textareaRef}
                  value={code}
                  onChange={handleCodeChange}
                  className="w-full h-full p-4 bg-transparent font-mono text-sm leading-6 resize-none focus:outline-none text-zinc-300 caret-white relative z-10"
                  spellCheck="false"
                  style={{ caretColor: "white" }}
                />
              </div>
            </div>
          ) : (
            <div className="flex-1 flex items-center justify-center text-zinc-500">
              <div className="text-center">
                <Coffee size={48} className="mx-auto mb-4 text-zinc-600" />
                <p className="text-lg">Welcome to Javaistic Code Editor</p>
                <p className="text-sm mt-2">
                  Open a file from the explorer to start coding
                </p>
              </div>
            </div>
          )}
        </div>
      </div>

      {/* Output Panel */}
      {output && (
        <div className="bg-zinc-800 rounded-b-2xl border-t border-zinc-700 h-48 flex flex-col">
          <div className="px-4 py-2 border-b border-zinc-700 text-sm font-medium flex items-center justify-between">
            <span>OUTPUT</span>
            <button
              onClick={() => setOutput("")}
              className="text-zinc-500 hover:text-white text-xs"
            >
              Clear
            </button>
          </div>
          <div className="flex-1 text-left p-4 font-mono text-sm overflow-auto rounded-b-2xl">
            <pre className="text-green-400 whitespace-pre-wrap">{output}</pre>
          </div>
        </div>
      )}
    </div>
  );
}
